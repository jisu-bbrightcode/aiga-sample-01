resolveDocumentEditorScreenBounds (lines 362-385):
```typescript
function resolveDocumentEditorScreenBounds(engine, nodeId) {
  const nodePos = engine.graph.getNodePosition(nodeId);
  if (!nodePos) return null;
  const surfaceX = nodePos.x + padding;
  const surfaceY = nodePos.y + padding;
  const topLeft = engine.flowToScreenPosition({ x: surfaceX, y: surfaceY });
  const bottomRight = engine.flowToScreenPosition({
    x: surfaceX + surfaceWidth, y: surfaceY + surfaceHeight,
  });
  return { left: topLeft.x, top: topLeft.y, width: ..., height: ... };
}
```
